\begin{frame}[shrink=10]
\frametitle{Beyond the Hype: Challenges of Using LLMs in Precision Medicine}

\begin{columns}[T]
\begin{column}{0.57\textwidth}
{\small
\textbf{Validity \& Reliability}
\begin{itemize}
\item LLMs can hallucinate, struggle with rare diseases, and may misinterpret subtle clinical signals.
\item Ensuring consistent, safe, and evidence-based outputs is essential.
\end{itemize}

\vspace{0.2cm}
\textbf{Data Privacy}
\begin{itemize}
\item Patient data is protected, fragmented, and heterogeneous---making it difficult to obtain large, high-quality datasets for training or fine-tuning securely.
\end{itemize}

\vspace{0.2cm}
\textbf{Ultra-High-Throughput Processing Requirements}
\begin{itemize}
\item Precision Medicine involves genomic, clinical, imaging, and longitudinal data at massive scale, requiring LLMs to handle multimodal, high-volume inputs efficiently.
\end{itemize}
}
\end{column}

\frametitle{Epil\={o}gus: Moving Toward Trustworthy and Effective AI in Precision Medicine}

\begin{columns}[T]

\begin{itemize}
\item Thinking LLMs (o1, DeepSeek-R1) bring new reasoning capabilities

\vspace{0.15cm}

\begin{itemize}

\item LA-MARRVEL: LLM-augmented context-aware genomic insights
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Faster \& Greener:}
\begin{itemize}
\item Efficient inference through sequential testing
\item Context engineering as a key performance multiplier
\end{itemize}

\vspace{0.15cm}
\textbf{Trustworthy:}
\begin{itemize}
\item Replication is not optional---rigorous validation matters
\item VALIDATE ASD: building reliable, reproducible AI workflows
\end{itemize}
\end{column}
